<script>

class InteractiveAlbum {

constructor() {

this.album = document.getElementById('album');

this.pages = document.querySelectorAll('.page');

this.prevBtn = document.getElementById('prevBtn');

this.nextBtn = document.getElementById('nextBtn');

this.currentPage = 0;

this.totalPages = this.pages.length / 2; // Каждый разворот - 2 страницы

this.init();

}

init() {

this.updatePages();

this.addEventListeners();

}

addEventListeners() {

this.prevBtn.addEventListener('click', () => this.prevPage());

this.nextBtn.addEventListener('click', () => this.nextPage());

// Добавляем поддержку клавиатуры

document.addEventListener('keydown', (e) => {

if (e.key === 'ArrowLeft') this.prevPage();

if (e.key === 'ArrowRight') this.nextPage();

});

// Добавляем свайпы на мобильных устройствах

let startX = 0;

this.album.addEventListener('touchstart', (e) => {

startX = e.touches[0].clientX;

});

this.album.addEventListener('touchend', (e) => {

const endX = e.changedTouches[0].clientX;

const diff = startX - endX;

if (Math.abs(diff) > 50) { // Минимальное расстояние свайпа

if (diff > 0) this.nextPage();

else this.prevPage();

}

});

}

nextPage() {

if (this.currentPage < this.totalPages - 1) {

this.currentPage++;

this.updatePages();

}

}

prevPage() {

if (this.currentPage > 0) {

this.currentPage--;

this.updatePages();

}

}

updatePages() {

// Сбрасываем все страницы

this.pages.forEach(page => {

page.style.zIndex = '0';

page.style.display = 'none';

});

// Показываем текущий разворот

const leftPageIndex = this.currentPage * 2;

const rightPageIndex = this.currentPage * 2 + 1;

if (this.pages[leftPageIndex]) {

this.pages[leftPageIndex].style.display = 'flex';

this.pages[leftPageIndex].style.zIndex = '2';

}

if (this.pages[rightPageIndex]) {

this.pages[rightPageIndex].style.display = 'flex';

this.pages[rightPageIndex].style.zIndex = '2';

}

// Обновляем состояние кнопок

this.prevBtn.disabled = this.currentPage === 0;

this.nextBtn.disabled = this.currentPage === this.totalPages - 1;

// Добавляем анимацию перелистывания

this.album.classList.add('flipped');

setTimeout(() => {

this.album.classList.remove('flipped');

}, 800);

}

}

// Инициализация альбома когда страница загружена

document.addEventListener('DOMContentLoaded', () => {

new InteractiveAlbum();

});

</script>
